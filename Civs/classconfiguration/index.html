    enabled: true
type: class
name: default # 与你的类型文件名称一致这里文件名 default.yaml 所以就用 default
icon: PLAYER_HEAD # 类选择菜单中的图标
is-in-shop: false # 如果不希望该类型在商店中有出售（可以通过前置条件进行控制）
level: 1 # 用于商店级别排序

children: [] # 必须解锁前置条件才可解锁该类型
pre-reqs:
  - skill:crafting=3 # 需要制作等级达到 Lv.3
  - potion_weakness:level=3 # 需要酿造等级达到 Lv.3
  - alchemist:level=21 # 需要炼金等级达到 Lv.21
  - hovel:level=2 # 需要一个 Lv.2 的小屋
  - member=hamlet:village:town:city:metropolis # 需要达到 小村庄或者更高规模的类型

mana-title: mana # key that corresponds to the translation file
mana-per-second: 1
max-mana: 50
max-health: 20
max-level: 8 # the max levels that can be put into this class

#允许操作的内容
allowed-actions:
  - BOW # allows the player to hold/use bows
  - CROSSBOW
  - TRIDENT
  - DIAMOND_SWORD
  - DIAMOND_AXE
  - DIAMOND_HELMET
  - DIAMOND_CHESTPLATE
  - DIAMOND_LEGGINGS
  - DIAMOND_BOOTS
  - SHARPNESS.5 # allows the player to hold/use items with sharpness 1, 2, 3, 4, or 5
  - PIERCING.5
  - POWER.5
  - PROTECTION.4

spells:
  1: # this slot is unlocked from the beginning
    - g:damage # allows any damage group spell to be used in this slot
    - g:disable # allows any disable group spell to be used in this slot
    - potion_weakness # allows the potion_weakness spell to be used in this slot
  2: # unlocked at level 4
    - g:buff
    - g:chase
    - g:heal
  3: # unlocked at level 8
    - g:escape
    - g:non-combat
